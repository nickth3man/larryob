
The best approach here is using **`pytest`** combined with an in-memory DuckDB database. This allows your tests to run instantly without altering your actual `nba_stats.db` file, and we will mock the NBA API response so the tests don't require an active internet connection or hit rate limits.

### 1. Setup for Testing

You will need to install `pytest` to run the suite.

```bash
pip install pytest

```

Assuming your main script from the previous step is saved as `scraper.py` in your `web-scraper` repository, you will create a new file named `test_scraper.py` right next to it.

### 2. The Test Suite (`test_scraper.py`)

This test suite covers three critical areas: ensuring the schema builds correctly, verifying the data transformation logic, and confirming that duplicate prevention (`INSERT OR IGNORE`) works as expected.

```python
import pytest
import duckdb
import pandas as pd
from unittest.mock import patch, MagicMock

# --- 1. Fixtures ---
@pytest.fixture
def memory_db():
    """Sets up an in-memory DuckDB instance for isolated testing."""
    con = duckdb.connect(':memory:')
    
    # Replicate the schema creation from the main script
    con.execute("""
    CREATE TABLE Player_Game_Logs (
        game_id VARCHAR, player_id VARCHAR, player_name VARCHAR, team_id VARCHAR,
        game_date DATE, matchup VARCHAR, wl VARCHAR, minutes_played DOUBLE,
        fgm INTEGER, fga INTEGER, fg3m INTEGER, fg3a INTEGER, ftm INTEGER,
        fta INTEGER, oreb INTEGER, dreb INTEGER, reb INTEGER, ast INTEGER,
        stl INTEGER, blk INTEGER, tov INTEGER, pf INTEGER, pts INTEGER,
        plus_minus INTEGER,
        PRIMARY KEY (game_id, player_id)
    );
    """)
    return con

@pytest.fixture
def mock_nba_data():
    """Provides a fake, standardized DataFrame mimicking the NBA API output."""
    data = {
        'GAME_ID': ['0022300001', '0022300002'],
        'PLAYER_ID': ['1628389', '203497'],
        'PLAYER_NAME': ['Bam Adebayo', 'Rudy Gobert'],
        'TEAM_ID': ['1610612748', '1610612750'],
        'GAME_DATE': ['2023-10-24', '2023-10-25'],
        'MATCHUP': ['MIA vs. DET', 'MIN @ TOR'],
        'WL': ['W', 'L'],
        'MIN': [32.5, 34.0],
        'FGM': [7, 5], 'FGA': [13, 8], 'FG3M': [0, 0], 'FG3A': [0, 0],
        'FTM': [5, 4], 'FTA': [6, 6], 'OREB': [2, 4], 'DREB': [8, 10],
        'REB': [10, 14], 'AST': [2, 1], 'STL': [1, 0], 'BLK': [2, 4],
        'TOV': [2, 1], 'PF': [3, 2], 'PTS': [19, 14], 'PLUS_MINUS': [8, -2]
    }
    return pd.DataFrame(data)

# --- 2. Tests ---

def test_schema_creation(memory_db):
    """Verifies that the table exists and has the correct columns."""
    result = memory_db.execute("PRAGMA table_info(Player_Game_Logs)").df()
    
    expected_columns = [
        'game_id', 'player_id', 'player_name', 'team_id', 'game_date', 
        'matchup', 'wl', 'minutes_played', 'fgm', 'fga', 'fg3m', 'fg3a', 
        'ftm', 'fta', 'oreb', 'dreb', 'reb', 'ast', 'stl', 'blk', 'tov', 
        'pf', 'pts', 'plus_minus'
    ]
    
    actual_columns = result['name'].tolist()
    assert all(col in actual_columns for col in expected_columns), "Schema is missing required columns."

def test_data_transformation_and_load(memory_db, mock_nba_data):
    """Tests the renaming and loading logic using mock data."""
    
    # 1. Transform (Simulating the logic in scraper.py)
    df_clean = mock_nba_data.rename(columns=lambda x: x.lower())
    
    # 2. Load into memory database
    memory_db.execute("INSERT INTO Player_Game_Logs SELECT * FROM df_clean")
    
    # 3. Assertions
    result = memory_db.execute("SELECT * FROM Player_Game_Logs").df()
    
    assert len(result) == 2, "Database should contain exactly two records."
    assert result.loc[result['player_name'] == 'Rudy Gobert', 'blk'].values[0] == 4, "Data mapping failed for blocks."

def test_insert_or_ignore_prevents_duplicates(memory_db, mock_nba_data):
    """Ensures running the script twice doesn't duplicate rows."""
    
    df_clean = mock_nba_data.rename(columns=lambda x: x.lower())
    
    # Load the same data twice
    memory_db.execute("INSERT OR IGNORE INTO Player_Game_Logs SELECT * FROM df_clean")
    memory_db.execute("INSERT OR IGNORE INTO Player_Game_Logs SELECT * FROM df_clean")
    
    # Count rows
    row_count = memory_db.execute("SELECT COUNT(*) FROM Player_Game_Logs").fetchone()[0]
    
    assert row_count == 2, "Primary key constraint failed; duplicates were inserted."

```

### 3. Running the Tests

To execute the suite, open your terminal, navigate to the folder containing these scripts, and simply run:

```bash
pytest test_scraper.py -v

```

The `-v` flag gives you verbose output, showing exactly which tests passed or failed.
